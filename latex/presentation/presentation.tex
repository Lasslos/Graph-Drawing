\begin{frame}{Introduction}
    Graphs are a prominent object in modern mathematics, useful for abstracting complex reallife situations.
    \begin{itemize}
        \item Railway networks, data structures, geometric objects
        \item Model relationships: vertices for entities, edges for connections
    \end{itemize}
\end{frame}

\begin{frame}
    \textbf{Example:} Two companies with employees as vertices and daily interactions as edges.
    \begin{itemize}
        \item Compare communication hierarchy and department interactions
        \item Difficulty: meticulously searching through huge lists of vertices and edges
    \end{itemize}
    \textbf{Goal:} Represent finite graphs in $\mathbb{R}^2$ and $\mathbb{R}^3$ esthetically.
    \begin{itemize}
        \item Drawing: vertices as points, edges as straight-line segments
        \item Quickly understand structure and type of a given graph
    \end{itemize}
\end{frame}

\begin{frame}{Graph embedding}
    \textbf{Graph definition:}
    \begin{itemize}
        \item set of vertices $V$, set of edges $E \subseteq \{ \{ u,v \} \mid u,v \in E\}$ then $G=(V,E)$
    \end{itemize}
    Our goal is to draw graphs, for our purposes a drawable graph has a finite set of vertices.
    \begin{itemize}
        \item finite $V \implies$ finite $E$
    \end{itemize}
    A graph embedding $p$ is an injective map $p : V \to A$, which maps every $v \in V$. $p$ induces an embedding of $E$, therefore $p$ is a lossless embedding of $G$ in $A$.
    \begin{itemize}
        \item $\exists p \iff |V| \leq |A|$
    \end{itemize}
    
    Let us call $H=p(G)$ and choose $A= \mathbb{R}^n, \quad n=2,3$
\end{frame}

\begin{frame}{Modeling stress}
    We add a drawing function $d : F \to \mathbb{R}, \{ a,b \} \mapsto \overline{ab}$, giving us a drawing of $H$.
    
    $W \subseteq \mathbb{R}^n$ essentially randomly chosen, drawings of $H$ are mostly tangled.
    We use a stress function $S$ that assigns the graph $H$ a real number, the stress in $H$.
    Minima of $S$ are our desired representations of $G$, if possible the global minimum as ideal.
\end{frame}

\begin{frame}{FR model}
    developed 1991, by Fruchterman and Reingold, following two main principles~\cite{fr_paper}:
    \begin{itemize}
    \item drawing neighboring vertices near each other
    \item vertices should not  be drawn too near to each other
    \end{itemize}

    The FR model uses the following stress-function:
    \begin{equation}
        S_{FR}(p(G)) = \frac{1}{6}\sum_{\{u,v\} \in E}{\parallel p(u)-p(v) \parallel^3}
        - \sum_{\{u,v \} \subseteq V}{\log \parallel p(u)-p(v) \parallel}
    \end{equation}
\end{frame}

\begin{frame}
    \textbf{Observe:}
    \begin{itemize}
        \item the further apart neighboring vertices get mapped the bigger the stress
        \item small distances between vertices let the logarithm explode
        \item the logarithm dominates at small distances and hinders the graph from collapsing, or converging into a single point
    \end{itemize}
    $S_{FR}$ satisfies the two main principles, while being rather quick when implemented into an algorithm.
\end{frame}

\begin{frame}{ERLL model}
    The Edge Repulsion LinLog model by Andreas Noack, 2003, places its focus on the graph's edges.
    For $v \in V$, let $\deg(v)$ denote the number of edges $v$ is an endpoint of~\cite{erll_paper}.
    \begin{equation}
    \begin{split}
        S_{ERLL}(p(G)) = & \frac{1}{2} \sum_{\{u,v\} \in E}{\parallel p(u)-p(v) \parallel} \\
        & - \sum_{\{u,v \} \subseteq V}{\deg(u)\deg(v) \log\parallel p(u)-p(v) \parallel}
    \end{split}
    \end{equation}
\end{frame}

\begin{frame}
    \textbf{Observe:}
    \begin{itemize}
        \item edges cause both attraction and repulsion
        \item $G$ of uniform degrees: FR $\approx$ ERLL scaled with $c=\deg{(v)}^2$
        \item $G$ of non-uniform degrees: FR repulsion remains the same, but vertices $u,v$ repel each other proportional to $\deg(u)\deg(v)$ in ERLL.
    \end{itemize}
    $\implies$ ERLL spreads out "big nodes" compared to FR, usefull in highly non-uniform graphs
\end{frame}

using Plots

function plot_graph(g::SimpleGraph, p::Matrix{Float64})
    plt = plot(legend = false, aspect_ratio = :equal)
    for e in edges(g)
        plot!(plt, [p[1, src(e)], p[1, dst(e)]],
            [p[2, src(e)], p[2, dst(e)]], color = :black)
    end
    scatter!(plt, p[1, :], p[2, :], markersize=5, color = :blue)
    return plt
end
\end{minted}

\subsection{Stress functions}

Since the two stress functions explored in the theoretical part is structured in a similar way,
we can implement them in a single function. We just need to pass the edge and vertex stress functions
as arguments. To do this, we use anonymous functions:

\begin{minted}{julia}
function stress(
    g::SimpleGraph, 
    edge_stress_f::Function, 
    vertex_stress_f::Function, 
    combine_f::Function)

    n = nv(g)
    edge_stress = 0
    for e in edges(g)
        edge_stress += edge_stress_f(src(e), dst(e))
    end
    vertex_stress = 0
    for i in 1:n
        for j in (i+1):n
            vertex_stress += vertex_stress_f(i, j)
        end
    end

    return combine_f(edge_stress, vertex_stress)
end
\end{minted}

Implementing the FR and ERLL stress functions is then straightforward:

\begin{minted}{julia}
function fr_stress(g::SimpleGraph, p::Matrix{Float64})
    D = pairwise(Euclidean(), p, dims=2)
    edge_stress_f = (i, j) -> D[i, j]^3
    vertex_stress_f = (i, j) -> log(D[i, j] + 10^(-6))
    combine_f = (x, y) -> x / 6 - y
    return stress(g, edge_stress_f, vertex_stress_f, combine_f)
end

function erll_stress(g::SimpleGraph, p::Matrix{Float64}, deg::Vector{Int})
    D = pairwise(Euclidean(), p, dims=2)
    edge_stress_f = (i, j) -> D[i, j]
    vertex_stress_f = (i, j) -> deg[i] * deg[j] * log(D[i, j] + 10^(-6))
    combine_f = (x, y) -> x / 2 - y
    return stress(g, edge_stress_f, vertex_stress_f, combine_f)
end
\end{minted}

Note that we add a small constant to the distance in the logarithm to prevent it
from exploding when two vertices are mapped to the same point.

We use the \texttt{pairwise} function from the Distances library~\cite{distancesjl} to compute the distance matrix $D$.
This is way more efficient than computing the distances in a nested loop.

Also note that the erll stress function takes an additional argument,
which is the degree of each vertex. This is used in the calculation,
but since the degree of each vertex does not change,
it would be inefficient to calculate it inside the vertex stress function.

\subsection{Gradient descent}

The final piece in the puzzle is to actually use the plotting and the stress functions
to find good embeddings of a given graph. To do this, we use gradient descent.
We use the Zygote library~\cite{zygotejl} to compute the gradients of the stress functions.

With zygote, we can easily differentiate through our stress functions,
which are implemented in a very straightforward way. Zygote does this by using source-to-source automatic differentiation,
which means that it transforms our code into a form that can be differentiated, and then applies the chain rule to compute the gradients.
This allows us to compute the gradients of our stress functions without having to manually derive them, which can be quite
tedious and error-prone.

The principle of gradient descent is the following:

We initialize the positions of the vertices randomly in a 2D space.
We then compute the stress and its gradient at each step, and update the positions by some rule.
To check for convergence, we either stop after the maximum number of steps has been reached,
or if the stress values do not change meaningfully.
It is important that we do not check for equality of the stress values, since they are floating-point numbers and can be very close
to each other without being exactly equal. Instead, we check if the relative difference is smaller than a small threshold.


\subsubsection{Simple gradient descent}
This is the simple version of the gradient descent algorithm. One can imagine what this algorithm does by imagining a hiker
being very tired and as such, trying to find the nearest town. To do that, he always goes in the direction of steepest descent.
Crucially, he does not have any momentum.

\begin{minted}{julia}
function simple_gradient_descent(g::SimpleGraph, stress_fn::Function, alpha::Float64,
    max_steps::Int64, filename::String, stress_fn_name::String)

    n = nv(g) # Number of vertices
    p = rand(2, n) # Initial positions

    stresses = Float64[]

    for k in 1:max_steps
        s, (grad,) = Zygote.withgradient(q -> stress_fn(g, q), p)
        push!(stresses, s)

        if length(stresses) > 1 &&
            abs((stresses[end] - stresses[end-1]) / stresses[end-1]) < 1e-7

            println("converged!")
            println("iteration ", k)
            plot1 = plot_graph(g, p)
            plot2 = plot(stresses, label="stress")
            final = plot(plot1, plot2, layout=2)
            savefig(final, filename * "_simple_" * stress_fn_name
                * "_a" * string(alpha) * "_i" * string(k) * ".svg")
            break
        end

        p -= alpha * grad

        if k % 30 == 0
            plot1 = plot_graph(g, p)
            plot2 = plot(stresses, label="stress")
            display(plot(plot1, plot2, layout=2))
        end
    end
end
\end{minted}

A couple notes:

\begin{itemize}
    \item Let $s_l$ be the last stress value and $s_{l-1}$ be the second to last stress value.
    To calulate if the algorithm converged, we do $\parallel \frac{s_l - s_{l-1}}{s_{l-1}} \parallel< \epsilon$.
    A more natural way might be $\parallel s_l - s_{l-1} \parallel < \epsilon$.
    However, with this approach, if the stress value is different for different functions, i.e.\ because the nodes are pushed further apart, non-visible changes would be in an order of magnitude higher,
    so to account for that, we should use the relative change instead of the absolute change.
    \item Again, we use anonymous functions here. In this case, to easily swap the stress function, it is passed as an argument instead of being hard-coded.
    \item The parameters \mintinline{julia}|filename::String| and \mintinline{julia}|stres_fn_name::String|
    are for easier management of multiple graph drawings at the same time. Earlier versions of that implementation just ouputted the same file, overwriting previous results.
\end{itemize}

\subsubsection{Nesterov's accelerated gradient descent}

In this version of gradient descent, we not only go a small step in the direction of the gradient, but we also add momentum to the term.

This can be imagined by a ball rolling, which in addition to going downhill, also has some inertia, which makes it roll
faster and faster as it goes down the hill. This can help to escape local minima and find better solutions.

Friction is modelled by $\beta$, which says how much of the previous momentum we keep. Because $\beta = 0.9$, we keep 90\% of the previous momentum.

\begin{minted}{julia}
function nesterov_gradient_descent(g::SimpleGraph, stress_fn::Function, alpha::Float64,
    beta::Float64, max_steps::Int64, filename::String, stress_fn_name::String)

    n = nv(g) # Number of vertices
    p = rand(2, n) # Initial positions

    stresses = Float64[]

    momentum = zeros(2, n)

    for k in 0:max_steps
        s, (grad,) = Zygote.withgradient(q -> stress_fn(g, q), p + beta * momentum)
        new_momentum = beta * momentum - alpha * grad
        push!(stresses, s)

        if length(stresses) > 50 &&
            abs((stresses[end] - stresses[end-1]) / stresses[end-1]) < 1e-7

            println("converged!")
            println("iteration ", k)
            plot1 = plot_graph(g, p)
            plot2 = plot(stresses, label="stress")
            final = plot(plot1, plot2, layout=2)
            savefig(final, filename * "_nesterov_" * stress_fn_name
                * "_a" * string(alpha) * "_i" * string(k) * ".svg")
            break
        end

        p += new_momentum
        momentum = new_momentum

        if k % 30 == 0
            plot1 = plot_graph(g, p)
            plot2 = plot(stresses, label="stress")
            display(plot(plot1, plot2, layout=2))
        end
    end
end
\end{minted}

\subsection{Figuring out good learning rates}
When trying our models, we had some problems with the suggested learning rates. For the FR stress, the suggested learning rates were fine,
but for ERLL, the learning rates were way to small, which made convergence very slow. Therefore, we analyzed which learning rate fits best for the
different stress values.

\subsubsection{Simple gradient descent, FR stress}
Let's start with the simple gradient descent method, fr stress, $\alpha=0.001$ and the wheel graph 11.

Almost always, the graph converges to something looking like Figure~\ref{wheel-graph-good}.

In a few cases (less than $10\%$), we converge into a local minimum like depicted in Figure~\ref{wheel-graph-bad}

%\paragraph{Turning up the learning rate}

When we increase the learning rate $\alpha$, there are interesting results.

\begin{figure}[H]
\centering 
\includesvg[width=0.7\textwidth]{../assets/wheel_graph_11_simple_fr_a0.01_i637.svg}
\caption{\label{wheel-graph-alpha-0.01} Wheel Graph 11, Simple Gradient Descent, $\alpha=0.01$, Took 428 Iterations}
\end{figure}

With a learning rate $\alpha=0.01$, the convergence speed dramatically increases with no visible poorer quality.

\begin{figure}[H]
\centering
\includesvg[width=0.7\textwidth]{../assets/wheel_graph_11_simple_fr_a0.1_i58.svg}
\caption{\label{wheel-graph-alpha-0.1} Wheel Graph 11, Simple Gradient Descent, $\alpha=0.1$, Took 57 Iterations}
\end{figure}

At $\alpha=0.1$, the convergence speed increases even more, with quality staying essentially the same.
However, the rate of poorly converged graphs increases, however this is difficult to quantify.
We also see that stress has actually increased twice, even though we were going ``downhill''.
This is because we are taking too big steps, which makes us overshoot the ``valley''.

\begin{figure}[H]
\centering
\includesvg[width=0.7\textwidth]{../assets/wheel_graph_11_simple_fr_a1.0_i11.svg}
\caption{\label{wheel-graph-alpha-1.0} Wheel Graph 11, Simple Gradient Descent, $\alpha=1.0$, Took 11 Iterations}
\end{figure}

Finally, at $\alpha=1.0$, the algorithm does not converge at all, and the stress values explode.
What's interesting is that it almost always takes 11 steps to diverge.
Investigating what is going on reveals that most of the parameters turn to $NaN$ at step 10, not changing at step 11 and therefore ``converging'' as per our definition.
The reason for this being essentially repeatable even though the initial positions are random
is because the initial positions are bound to be in a range between 0 and 1.

From this, we can learn that we should always be careful when choosing the learning rate.
For a simple graph, there are not as many local minima, so bigger learning rates still work,
but we should not do this for a more complex graph.

\subsubsection{Simple gradient descent, ERLL stress}

Continuing with the simple gradient descent method, erll stress, $\alpha=0.001$ and the wheel graph 11,
we see that convergence is painfully slow. On the graph, we can see the vertices spreading out
way more than with fr stress. This reveals a fundamental difference:
ERLL stress allows vertices to be far away more than FR stress, which makes the learning rate proportionally smaller.
We increase the learning rate to $\alpha=0.1$ and see that convergence is much faster.

\centering
\includesvg[width=0.7\textwidth]{../assets/wheel_graph_11_simple_erll_a0.1_i20994.svg}
\caption{\label{wheel-graph-erll} Wheel Graph 11, Simple Gradient Descent, $\alpha=0.1$, Took 20994 Iterations}

%\paragraph{A good learning rate}

Considering that at $\alpha=0.001$ compute times are reasonable for FR Stress,
and at $\alpha=0.1$ compute times are reasonable for ERLL Stress, we can say that these are good learning rates for the respective stress functions.

For simplicity reasons, we will use the same $\alpha$ for the nesterov gradient descent,
one could of course tune this separately.

\section{Conclusion}

In this project, we have explained the theory behind graph drawing,
implemented the graph drawing algorithm in Julia and compared different approaches and settings.

In our implementation, FR stress is not only much faster, but also produces better results.
ERLL stress just pushes vertices too far apart.

With the settings above, we have created some nice drawings of graphs.
They offer true insight into the structure even with no knowledge of the graphs themselves.

However, some graphs like the circular ladder graph almost always get stuck in suboptimal positions,
because they are planar in theory but in practice, the algorithm cannot untwist the two circles.

Graphs like bipartite graphs also get drawn in a very unsual way with this method.

While fun, for a truly good graph drawing algorithm, many improvements are possible.
